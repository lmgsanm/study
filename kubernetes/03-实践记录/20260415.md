yaml











```
stages:
  - test
  - build
  - scan
  - push
  - deploy

# 全局变量（你只需要修改这里）
variables:
  # 镜像仓库地址（可以是 GitLab 自带仓库、Harbor、Docker Hub）
  IMAGE_PREFIX: "registry.example.com/your-project"
  # GitLab 地址
  GITLAB_URL: "http://gitlab.example.com"
  # ArgoCD 地址
  ARGOCD_SERVER: "argocd.example.com"
  # 项目名称

# 缓存加速

  paths:
    - frontend/node_modules
    - backend/go/pkg

# ====================== 1. 代码扫描 ======================
code_scan:
  stage: test
  image: golang:1.22-alpine
  script:
    - echo "=== 代码扫描（go vet + 静态检查）==="
    - cd backend
    - go vet ./...
    - echo "代码扫描完成 ✅"
  only:
    - main
    - dev
  tags:
    - my-k8s-runner

# ====================== 2. 构建前后端镜像 ======================
build_frontend:
  stage: build
  image: docker:24-dind
  services:
    - docker:24-dind
  script:
    - echo "=== 构建 Frontend 镜像 ==="
    - docker build -t ${IMAGE_PREFIX}/frontend:${CI_COMMIT_SHORT_SHA} -f frontend/Dockerfile frontend/
    - docker tag ${IMAGE_PREFIX}/frontend:${CI_COMMIT_SHORT_SHA} ${IMAGE_PREFIX}/frontend:latest
  only:
    - main
    - dev
  tags:
    - my-k8s-runner

build_backend:
  stage: build
  image: docker:24-dind
  services:
    - docker:24-dind
  script:
    - echo "=== 构建 Backend 镜像 ==="
    - docker build -t ${IMAGE_PREFIX}/backend:${CI_COMMIT_SHORT_SHA} -f backend/Dockerfile backend/
    - docker tag ${IMAGE_PREFIX}/backend:${CI_COMMIT_SHORT_SHA} ${IMAGE_PREFIX}/backend:latest
  only:
    - main
    - dev
  tags:
    - my-k8s-runner

# ====================== 3. 镜像安全扫描（Trivy）======================
scan_frontend:
  stage: scan
  image: aquasec/trivy:latest
  script:
    - echo "=== 扫描 Frontend 镜像 ==="
    - trivy image --severity HIGH,CRITICAL --exit-code 1 ${IMAGE_PREFIX}/frontend:${CI_COMMIT_SHORT_SHA}
  only:
    - main
    - dev
  tags:
    - my-k8s-runner

scan_backend:
  stage: scan
  image: aquasec/trivy:latest
  script:
    - echo "=== 扫描 Backend 镜像 ==="
    - trivy image --severity HIGH,CRITICAL --exit-code 1 ${IMAGE_PREFIX}/backend:${CI_COMMIT_SHORT_SHA}
  only:
    - main
    - dev
  tags:
    - my-k8s-runner

# ====================== 4. 推送镜像 ======================
push_frontend:
  stage: push
  image: docker:24-dind
  services:
    - docker:24-dind
  script:
    - echo "=== 推送 Frontend 镜像 ==="
    - docker login -u ${CI_REGISTRY_USER} -p ${CI_REGISTRY_PASSWORD} ${CI_REGISTRY}
    - docker push ${IMAGE_PREFIX}/frontend:${CI_COMMIT_SHORT_SHA}
    - docker push ${IMAGE_PREFIX}/frontend:latest
  only:
    - main
  tags:
    - my-k8s-runner

push_backend:
  stage: push
  image: docker:24-dind
  services:
    - docker:24-dind
  script:
    - echo "=== 推送 Backend 镜像 ==="
    - docker login -u ${CI_REGISTRY_USER} -p ${CI_REGISTRY_PASSWORD} ${CI_REGISTRY}
    - docker push ${IMAGE_PREFIX}/backend:${CI_COMMIT_SHORT_SHA}
    - docker push ${IMAGE_PREFIX}/backend:latest
  only:
    - main
  tags:
    - my-k8s-runner

# ====================== 5. ArgoCD 自动发布 ======================
deploy_argocd:
  stage: deploy
  image: argoproj/argocd:latest
  script:
    - echo "=== ArgoCD 同步应用 ==="
    - argocd login ${ARGOCD_SERVER} --username=admin --password=${ARGOCD_PASSWORD} --insecure
    - argocd app sync ${PROJECT_NAME}
    - argocd app wait ${PROJECT_NAME} --health
    - echo "应用发布完成 ✅"
  only:
    - main
  tags:
    - my-k8s-runner
```

```
  name: go-app

  project: default

    targetRevision: main
    path: k8s/

    namespace: default
  syncPolicy:

      selfHeal: true
      prune: true
```

、itOps 标准模式（CI 改 Git + Argo 自动同步）

**适合生产、最稳定**：CI 只构建镜像，然后提交新 tag 到 Git 清单仓库；Argo CD 监听 Git 自动同步。

### 1. GitLab CI 流水线：构建镜像 → 提交新 tag 到 Git

#### 步骤

1. 准备 **Git 清单仓库**（存放 deployment.yaml/values.yaml）
2. CI 构建推送镜像后，**自动修改 YAML 里的 image.tag**
3. 提交并推送到 Git → Argo CD 检测到变更 → 自动同步

#### 示例 .gitlab-ci.yml（关键片段）

yaml











```
# 变量（沿用你之前内部地址配置）
variables:
  CI_REGISTRY: "gitlab-webservice-default.gitlab.svc:8080"
  CI_REGISTRY_IMAGE: "${CI_REGISTRY}/${CI_PROJECT_PATH}"
  IMAGE_TAG: "${CI_REGISTRY_IMAGE}:${CI_COMMIT_SHORT_SHA}"
  # Git 清单仓库信息（你的 K8s 配置仓库）
  MANIFEST_REPO: "http://gitlab-webservice-default.gitlab.svc:8080/devops/k8s-manifests.git"
  MANIFEST_PATH: "demo-app/frontend/deployment.yaml" # 要修改的文件

# 阶段
stages:
  - build
  - push
  - update-manifest

# 1. 构建推送镜像（你已有的）
build-push:
  stage: build
  image: docker:24-dind
  services:
    - docker:24-dind
  script:
    - echo "$CI_REGISTRY_PASSWORD" | docker login "$CI_REGISTRY" -u "$CI_REGISTRY_USER" --password-stdin
    - docker build -t $IMAGE_TAG .
    - docker push $IMAGE_TAG

# 2. 关键：自动更新 Git 清单里的镜像 tag
update-manifest:
  stage: update-manifest
  image: alpine/git:latest
  before_script:
    # 配置 Git 身份
    - git config --global user.name "GitLab CI"
    - git config --global user.email "ci@example.com"
    # 克隆清单仓库（用内部地址+CI Token）
    - git clone http://gitlab-ci-token:${CI_JOB_TOKEN}@${MANIFEST_REPO#http://} manifests
    - cd manifests
  script:
    # 用 sed 替换 deployment.yaml 里的 image 行（适配你的 YAML）
    - sed -i "s|image:.*|image: ${IMAGE_TAG}|g" ${MANIFEST_PATH}
    # 提交推送
    - git add ${MANIFEST_PATH}
    - git commit -m "ci: update demo-app image to ${CI_COMMIT_SHORT_SHA}"
    - git push origin main
```



```
Argo CD Image Updater（直接监控镜像仓库）
适合测试 / 开发环境：无需 CI 修改 Git，Image Updater 直接轮询镜像仓库，发现新 tag 就通知 Argo CD 更新。
1. 安装 Image Updater
bash
运行
kubectl apply -n argocd -f https://raw.githubusercontent.com/argoproj-labs/argocd-image-updater/stable/manifests/install.yaml
2. 配置私有镜像仓库（GitLab Registry）
创建 ConfigMap 配置镜像仓库认证：
yaml
apiVersion: v1
kind: ConfigMap
metadata:
  name: argocd-image-updater-config
  namespace: argocd
data:
  registries.conf: |
    - name: gitlab-registry
      api_url: http://gitlab.example.com # 你的 GitLab Registry
      prefix: gitlab.example.com
      credentials: pullsecret
      pullSecret: gitlab-registry-secret # 存放 Registry 账号密码的 Secret
      default: true
创建 Registry 认证 Secret：
bash
运行
kubectl create secret docker-registry gitlab-registry-secret \
  -n argocd \
  --docker-server=gitlab.example.com \
  --docker-username=gitlab-ci-token \
  --docker-password=${CI_REGISTRY_PASSWORD}
3. 给 Argo CD Application 加注解（核心）
在 Application 上添加注解，告诉 Image Updater 监控哪个镜像
yaml
apiVersion: argoproj.io/v1alpha1
kind: Application
metadata:
  name: demo-app
  namespace: argocd
  annotations:
    # 核心：监控镜像 + 版本规则
    argocd-image-updater.argoproj.io/image-list: frontend=gitlab.example.com/devops/demo:~v1.0
    # 更新策略：semver（语义化版本）/newest-build（最新构建）/digest（追踪latest）
    argocd-image-updater.argoproj.io/frontend.update-strategy: semver
    # 允许的 tag 正则（只匹配 vx.x.x）
    argocd-image-updater.argoproj.io/frontend.allow-tags: regexp:^v[0-9]+\.[0-9]+\.[0-9]+$
    # 自动同步
    argocd-image-updater.argoproj.io/write-back-method: argocd
spec:
  # 其余配置同上 + 开启自动同步
  syncPolicy:
    automated:
      prune: true
      selfHeal: true
常用更新策略：
semver：语义化版本（推荐生产，如 v1.0.0 → v1.0.1）
newest-build：按构建时间选最新（适合短 SHA 标签，如 CI_COMMIT_SHORT_SHA）
digest：追踪 latest 标签（只要镜像摘要变就更新）
效果：
Image Updater 每 2 分钟轮询镜像仓库
发现新 tag（符合版本规则）→ 通知 Argo CD
Argo CD 自动更新应用镜像 → 滚动发布
```



Argo CD 本身不直接实现灰度（金丝雀）/ 蓝绿发布，必须配合 **Argo Rollouts**（Argo 生态的渐进式交付 CRD）来完成。下面给你 **生产可用、可直接复制** 的完整方案（金丝雀 + 蓝绿）。

## 一、核心原理

- **Argo CD**：负责 Git 同步、版本管理、应用生命周期

- 流量控制

  - 简单场景：按 **副本比例** 分流（无需服务网格）
  - 精细场景：**Istio/Linkerd/Nginx Ingress** 权重分流

## 三、方案 1：金丝雀发布（Canary，推荐灰度）

**场景**：逐步切流量（10% → 50% → 100%），支持暂停、人工确认、指标自动回滚。

### 1. Rollout YAML（替换 Deployment）

```
rollout-demo.yaml
```

yaml











```
apiVersion: argoproj.io/v1alpha1
kind: Rollout  # 替代 Deployment
metadata:
  name: demo-app
  namespace: dev
spec:
  replicas: 4
  selector:
    matchLabels:
      app: demo-app
  template:
    metadata:
      labels:
        app: demo-app
    spec:
      containers:
      - name: demo-app
        image: gitlab.example.com/devops/demo:v1  # 初始版本
        ports:
        - containerPort: 8080
        readinessProbe:
          httpGet:
            path: /health
            port: 8080
          initialDelaySeconds: 5
          periodSeconds: 10

  # --------------------------
  # 核心：金丝雀策略
  # --------------------------
  strategy:
    canary:
      # 分步发布
      steps:
      - setWeight: 10       # 第1步：10%流量到新版本
      - pause: {duration: 300s}  # 暂停5分钟观察
      - setWeight: 50       # 第2步：50%流量
      - pause: {duration: 300s}  # 再暂停5分钟
      - setWeight: 100      # 第3步：全量
      
      # 可选：最大不可用/溢出
      maxSurge: 25%
      maxUnavailable: 0

      # 可选：结合 Prometheus 自动分析（生产必备）
      analysis:
        templates:
        - templateName: demo-app-analysis
        args:
        - name: service-name
          value: demo-app
```

```
  strategy:
```
