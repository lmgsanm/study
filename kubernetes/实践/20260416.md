  --set global.hosts.gitlab.host=gitlab.example.com \
  --set global.hosts.domain=example.com \
  --set global.ingress.tls.enabled=false \
  --set certmanager.install=false \
  --set gitlab.webservice.ingress.annotations."nginx.ingress.kubernetes.io/ssl-redirect"=false \
  --set gitlab.webservice.workhorse.enabled=true

```
runners:
  config: |
    [[runners]]
      [runners.kubernetes]
        namespace = "{{.Release.Namespace}}"
        image = "ubuntu:18.04"
        privileged = true
        [[runners.kubernetes.volumes.host_path]]
          name = "docker"
          mount_path = "/var/run/docker.sock"
          
```



```
[runners.kubernetes]
  privileged = false # Usually works without privileged if permissions are correct
  [[runners.kubernetes.volumes.host_path]]
    name = "docker-sock"
    mount_path = "/var/run/docker.sock"
    host_path = "/var/run/docker.sock"
```
