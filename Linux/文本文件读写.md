# 写纯文本文件

### 示例1

```
package main

import (
	"bufio"
	"fmt"
	"os"
)

func main() {
	filename := "test.txt"
	file, err := os.OpenFile(filename, os.O_WRONLY|os.O_CREATE, 0666)
	if err != nil {
		fmt.Printf("打开文件错误=%v \n", err)
		return
	}
	defer file.Close()

	str := "http://lmgsanm.com \n"
	writer := bufio.NewWriter(file)
	for i := 0; i < 4; i++ {
		writer.WriteString(str)
	}
	writer.Flush()
}
```

示例1结果

```
http://lmgsanm.com 
http://lmgsanm.com 
http://lmgsanm.com 
http://lmgsanm.com 
```

示例：读取文本文件

```
package main

import (
	"bufio"
	"fmt"
	"io"
	"os"
)

func main() {
	filename := "test.txt"
	file, err := os.Open(filename)
	if err != nil {
		fmt.Println("打开文件失败= ", err)
	}
	defer file.Close()

	reader := bufio.NewReader(file)
	for {
		str, err := reader.ReadString('\n')
		if err == io.EOF {
			break
		}
		fmt.Println(str)
	}
	fmt.Println("文件读取完毕....")
}

```

示例结果

```
http://lmgsanm.com 

http://lmgsanm.com 

http://lmgsanm.com 

http://lmgsanm.com 

文件读取完毕....
```
